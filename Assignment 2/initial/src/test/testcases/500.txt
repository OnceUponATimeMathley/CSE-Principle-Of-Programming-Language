
Function: main
    Body:
        Var: sum = 0, a = 1;
        While a < 10 Do
            Var: b = 1, prod = 1;

                prod = prod * b;

            EndWhile.
            sum = sum + prod;
            a = a + 1;
        EndWhile.
    EndBody.
