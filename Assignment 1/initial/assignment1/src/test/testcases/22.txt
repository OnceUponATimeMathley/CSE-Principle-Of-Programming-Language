
		Function: sort
		Parameter: a,n
		Body:
			Var: i,j;
			For (i = 0, i < n-1, i + 1) Do
				For (j = i + 1, j < n, j + 1 ) Do
					If a[i] > a[j] Then 
						Var: t;
						t = a[i];
						a[i] = a[j];
						a[j] = t;
					EndIf.
				EndFor.
			EndFor.
			Return 1;
		EndBody.
